    3 切片

    注意要点：

    1. The slice is a reference to the array, so the slice is the reference type, obeying the mechanism of reference transmission.
    2. The use of sections is similar to the array, including traversal, accessing section elements, and seeking section length len (slice).
    3. The length of the slice can vary, and the slice can change dynamically.
    4. Basic syntax for the slice definition: 
    //as: var slice name [] type 
var a [] int

    实例练习

    package main

import "fmt"

func updateSlice(s []int){
    s[0] = 100
}
func main() {
    arr := [...] int{0,1,2,3,4,5,6,7}
    fmt.Println("arr[2:6] =",arr[2:6])
    s1 := arr[2:]
    fmt.Println("arr[:6] =",arr[:6])
    s2 := arr[:]
    fmt.Println("arr[2:] =",arr[2:])
    fmt.Println("arr[:] =",arr[:])
    updateSlice(s1)
    updateSlice(s2)
    fmt.Println(s1)
    fmt.Println(s2)

    fmt.Println("Extending slice")
    arr[0],arr[2] = 0, 2
    s1 = arr[2:6]
    s2 = s1[3:5]
    //slice有三个变量ptr.cap,len
    //slice可以向后扩展，不可以向前扩展
    //s[i]不可以超越len(s)，向后不能超过capacity
    s3 := append(s2,10)
    s4 := append(s3,11)
    s5 := append(s3,12)
    fmt.Println("s3,s4,s5 = ",s3,s4,s5)
    fmt.Println("arr = ",arr)

}

    package main

import "fmt"
//查看len 和 capacity
func printSlice(s []int){
    fmt.Printf("%v, len=%d, cap=%d\n",s, len(s), cap(s))
}

func main() {
    var s []int //Zero value for slice is nil
    for i := 0 ; i < 100; i++{
        printSlice(s)
        s = append(s,2 * i +1)
    }
    fmt.Println(s)
    s1 := []int{2,4,6,8}
    printSlice(s1)

    s2 := make([]int,16) //建立长度为16的slice
    s3 := make([]int,10,32)//len = 10,cap = 32
    printSlice(s2)
    printSlice(s3)
    fmt.Println("Copying slice")
    copy(s2,s1)
    printSlice(s2)
    //用切片的[x：y]来删除数组的元素
    fmt.Println("Deleting elements from slice")
    s2 = append(s2[:3], s2[4:]...)//删除第四个数字8
    printSlice(s2)

    fmt.Println("Popping from front")
    front := s2[0]
    s2 = s2[1:]
    fmt.Println("Popping from back")
    tail := s2[len(s2)-1]
    s2 = s2[:len(s2) - 1]
    fmt.Println(front)
    fmt.Println(tail)
    printSlice(s2)
}
