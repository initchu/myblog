    1 声明常变量¶

    Golang 变量使用的三种方式¶

    第一种：指定变量类型，声明后若不赋值，使用默认值；¶

     var i int

    第二种：根据值自行判定变量类型(类型推导）；¶

    var num = 10.11

    第三种：省略var,注意 := 左侧的变量不应该是已经声明过的，否则会导致编译错误；¶

    name := "tom"

    多变量声明在编程中，有时我们需要一次性声明多个变量，Golang也提供这样的语法；

    var n1,n2,n3 int
var n1,name,n3 = 100,"tom",888
n1,name,n3 := 100,"tom",888

    实例练习：

    package main

import (
    "fmt"
    "math"
    "math/cmplx"
)
var aa = 3
var bb = true
var ss = "kkk"//包内变量
var (
    aaa = 3
    sss = "kkk"
    bbb = true
)
func variableZeroValue(){
    var a int
    var s string
    fmt.Println(a,s)//Println只能放变量
    fmt.Printf("%d %q\n",a,s)

}
func variableInitialValue(){
    var a,b int = 3,4
    var s string = "abc"
    fmt.Println(a,b,s)
}

func variableTypeDeduction(){
    var a,b,c,s = 3, 4, true, "def"
    fmt.Println(a,b,c,s)
}
func variableShorter(){
    a,b,c,s := 3,4,true,"def"//:=只能在函数内使用，不需要var
    b =5
    fmt.Println(a,b,c,s)
}
func euler(){
    fmt.Printf("%.3f",
        cmplx.Pow(math.E, 1i * math.Pi) + 1)
}
func triangle(){
    var a,b int = 3,4
    var c int
    c = int(math.Sqrt(float64(a*a+b*b)))
    fmt.Println(c)
}

func consss() {
    const filename = "abc.txt"
    const a, b = 3, 4
    var c int
    c = int(math.Sqrt(a*a + b*b))
    fmt.Println(filename, c)
}
func enums(){
    const(
        cpp = 0
        java = 1
        python = 2
    )
    fmt.Println(cpp,java,python)
    const(
        b = 1 << (10*iota)//iota作为自增值
        kb
        mb
        gb
        tb
        ob
    )
    fmt.Println(b,kb,mb,gb,tb,ob)
}
func main() {
    fmt.Println("Hello world")
    variableInitialValue()
    variableZeroValue()
    variableTypeDeduction()
    euler()
    triangle()
    consss()

}

    Golang常量的使用方法¶

    package main
func main(){
    var num int
    const tax int = 0 //常量在声明的时候必须给赋值
    fmt.Println(num,tax)
}
    1. 常量在声明的时候必须给赋值
    2. 常量不能修改
    3. 常量只能修饰bool，数值类型，string 类型

    语法: **const identifier [type] = value** value可以是一个表达式比如 9/3

    const(
    a = 1
    b = 2
)

    const(
    a = iota //0
    b //1
    c //2
)
